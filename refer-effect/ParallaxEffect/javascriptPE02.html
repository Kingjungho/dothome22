document.querySelectorAll(".paraDot ul li a").forEach((el) => {
el.addEventListener("click", (e) => {
        e.preventDefault();
        document.querySelector(el.getAttribute("href")).scrollIntoView({
            behavior:"smooth"
        })
    })
})

window.addEventListener("scroll", () => {
    let scrollTop = document.documentElement.scrollTop || window.scrollY || window.pageYOffset;

    document.querySelector(".paraScroll span").innerText = parseInt(scrollTop) + "px";

    if( scrollTop > document.getElementById("section1").offsetTop ) {
        document.querySelectorAll(".paraDot ul li").forEach((el) => {
            el.classList.remove("active");
        })
        document.querySelector(".paraDot ul li:nth-child(1)").classList.add("active");
        }
    });
