.mouseCont {
    display: flex;
    align-items: center;
    justify-content: center;
    flex-direction: column;
    color: rgb(255, 255, 255);
    height: 100vh;
    width: 100%;
    overflow: hidden;
    font-family: 'NEXONLv2Gothic';
    cursor: none;
}
.mouseCont p {
    font-size: 2.5vw;
    line-height: 2;
    font-weight: 300;
}
.mouseCont h2 {
    font-size: 3vw;
    font-weight: normal;
    font-weight: 400;
    line-height: 1.5;
}
.mouseCont em {
    font-style: normal;
    border-bottom: 0.21vw dashed rgb(212, 212, 212);
    color: rgb(243, 45, 45);
}
.cursor {
    position: absolute;
    left: 0; top: 0;
    width: 10px;
    height: 10px;
    border-radius: 50%;
    z-index: 9999;
    background-color: #fff;
    user-select: none; 
    /* 컨트롤이 불가능 */
    pointer-events: none;
    border: 1px solid #fff;
    transition: transform 0.3s;
    mix-blend-mode: difference;
}
.cursor.active {
    transform: scale(15);
    background-color: rgba(255, 255, 255, 0.432);
}
.info.left.list li.active {
    text-decoration: underline;
    color: rgb(243, 45, 45);
}


const w = $(".cursor").width() / 2;
const h = $(".cursor").height() / 2;
//움직임 효과
$(window).mousemove(function(e){
    gsap.to(".cursor", {duration: 0.4, left: e.pageX - 5, top: e.pageY - 5});
});

//오버 효과
$(".mouseCont em").hover(function(){
    $(".cursor").addClass("active");
}, function(){
    $(".cursor").removeClass("active");
});

//출력용
$(".info.left.list li").click(function(){
    $(".info.left.list li").removeClass();
    $(this).addClass("active");
    $(".cursor").css("mix-blend-mode", $(this).text());
});
