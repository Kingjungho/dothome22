{

        this.name = name;

            document.write(this.num + ".내 이름은 " + this.name + "이며, 직업은" + this.job + "입니다.");

    const info2 = new func("2", "웹스토리보이", " 프론트앤드 개발자"); //인스턴스 생성

    info1.result();
    info2.result();
