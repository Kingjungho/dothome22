const scrollMenu = function(){
    document.querySelectorAll(".paraNav ul li a").forEach(li => {
        li.addEventListener("click", e => {
            e.preventDefault();
            document.querySelector(li.getAttribute("href")).scrollIntoView({
                behavior: "smooth"
            })
            })
    });
}
